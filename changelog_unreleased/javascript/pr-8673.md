#### Allow more arrow function print on same line with arrow ([#8673](https://github.com/prettier/prettier/pull/8673) by [@fisker](https://github.com/fisker))

<!-- prettier-ignore -->
```js
// Input
() => foo({
key: value});

() => new Foo({
key: value});

a= [() => foo({
key: value}), second];

a = {b: () => foo({
key: value})};

function a() {return () => foo({
key: value});}

// Prettier stable
() =>
  foo({
    key: value,

  new Foo({

      key: value,

];

    foo({
      key: value,
    }),
};

    foo({
      key: value,
    });
}

() => foo({
  key: value,

  key: value,
});
```
